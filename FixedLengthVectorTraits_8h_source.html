44 static std::vector<UnderlyingType_t<T>, Allocator> read(
45 const void *buffer, const H5::DataType &dtype) {
46 if constexpr (has_static_h5dtype_v<T>) {
47 H5::ArrayType arrType = dtype.getId();
48 std::vector<hsize_t> dims = getArrayDims(arrType);
49 if (dims.size() != 1)
50 throw std::invalid_argument("Unexpected number of array dimensions!");
51 std::size_t nElems = dims[0];
52 H5Buffer tmp;
53 H5::DataType targetSuper = getH5DType<T>();
54 if (arrType.getSuper() != targetSuper) {
55 // Need to convert
56 tmp = convert(buffer, dtype, H5::ArrayType(targetSuper, 1, dims.data()));
57 buffer = tmp.get();
58 }
59 if constexpr (std::is_trivial_v<UnderlyingType_t<T>>) {
60 const UnderlyingType_t<T> *start =
61 reinterpret_cast<const UnderlyingType_t<T> *>(buffer);
62 return std::vector<UnderlyingType_t<T>, Allocator>(start, start + nElems);
63 } else {
64 std::vector<UnderlyingType_t<T>, Allocator> vec;
65 vec.reserve(nElems);
66 const std::byte *byteBuffer = static_cast<const std::byte *>(buffer);
67 for (std::size_t idx = 0; idx < nElems; ++idx)
68 vec.push_back(fromBuffer<UnderlyingType_t<T>>(
69 byteBuffer + idx * targetSuper.getSize(), targetSuper));

71 }

74 std::vector<UnderlyingType_t<T>> out;

89 // Check the data type

91 std::memcpy(buffer, value.data(), sourceDType.getSize());

95 // Now copy this
96 std::memcpy(buffer, converted.get(), dtype.getSize());
97 converted.transferVLenOwnership().release();
98 }
99 } else {
100 std::byte *byteBuffer = static_cast<std::byte *>(buffer);
101 H5::ArrayType arrType = dtype.getId();
102 H5::DataType targetSuper = arrType.getSuper();

104 if (dims.size() != 1)
