21 template <template <typename> typename Op>
22 H5Buffer parallelMerge(const std::vector<std::pair<H5::DataType, const void *>> &buffers) {
23 std::vector<H5::DataType> dtypes;
24 dtypes.reserve(buffers.size());
25 for (const auto &p : buffers)
26 dtypes.push_back(p.first);
27 H5::DataType common = getCommonDType(dtypes);
28 std::vector<H5Buffer> owning;
29 std::vector<const void *> converted;
30 owning.reserve(buffers.size());
31 converted.reserve(buffers.size());
32 for (const auto &p : buffers) {
33 if (p.first == common)
34 // No need to convert
35 converted.push_back(p.second);
36 else {
37 owning.push_back(convert(p.second, p.first, common));
38 converted.push_back(owning.back().get());
39 }
40 }
41 switch (common.getClass()) {
42 case H5T_INTEGER:
43 case H5T_FLOAT:
44 case H5T_BITFIELD:
45 return parallelMerge<Op>(getNativeNumericDType(common), converted);
46 case H5T_ARRAY:
47 if (!isNumericDType(common.getSuper()))
48 throw std::invalid_argument(
49 "Common type " + to_string(common) + " does not have a numeric super type");
50 return parallelMerge<Op>(H5::ArrayType(common.getId()), converted);
51 default:
52 throw std::invalid_argument(
53 "parallelMerge can only handle numeric types and arrays of numeric types, "
54 "not " +
55 to_string(common));
56 return H5Buffer();
57 }

61 template <
62 template <typename> typename Operator, template <typename> typename Identity,
63 template <typename> typename Condition>
64 H5Buffer ParallelMerger<Operator, Identity, Condition>::mergeBuffers(
65 const std::vector<std::pair<H5::DataType, const void *>> &buffers) {
66 // Get the common data type
67 std::vector<H5::DataType> dtypes;
68 dtypes.reserve(buffers.size());
69 for (const auto &p : buffers)
70 dtypes.push_back(p.first);
71 H5::DataType common = getCommonDType(dtypes);
72 // Convert all the input buffers to the correct data type
73 // If something needs to be converted something needs to own that new memory until
74 // the end of this function
75 std::vector<H5Buffer> owning;
76 std::vector<const void *> converted;
77 owning.reserve(buffers.size());
78 converted.reserve(buffers.size());
79 std::set<std::size_t> vlenSizes;
80 for (const auto &p : buffers) {
81 if (p.first == common)
82 // No need to convert
83 converted.push_back(p.second);
84 else {
85 owning.push_back(convert(p.second, p.first, common));
86 converted.push_back(owning.back().get());
87 }
88 if (common.getClass() == H5T_VLEN) {
89 // If it's a variable length data type we need to get the pointers to its memory
90 // and get the number of values in each instance
91 // Get the stored VLen data
92 const hvl_t *vlenData = static_cast<const hvl_t *>(converted.back());
93 vlenSizes.insert(vlenData->len);
94 converted.back() = vlenData->p;
95 }
96 }
97 if (isNativePredefinedDType(common))
98 return H5Buffer(
99 apply_if<Condition, Functor>(
100 getNativePredefinedDType(common), 1, converted),
101 common);
102 else {
103 if (!isNativePredefinedDType(common.getSuper()))
104 throw std::invalid_argument(
105 "Common type " + to_string(common) +
106 " does not have a native predefined super type");
107 H5::PredType commonSuper = getNativePredefinedDType(common.getSuper());
108 if (common.getClass() == H5T_ARRAY) {
109 std::size_t n = getNArrayElements(common.getId());
110 return H5Buffer(
111 apply_if<Condition, Functor>(commonSuper, n, converted), common);
112 } else if (common.getClass() == H5T_VLEN) {
113 if (vlenSizes.size() != 1)
114 throw std::invalid_argument("Different sizes for variable length data!");
115 std::size_t n = *vlenSizes.begin();
116 return H5Buffer(
117 apply_if<Condition, Functor>(commonSuper, n, converted), common);
118 } else
119 throw std::invalid_argument(
120 "ParallelMerger can only handle predefined types and "
121 "arrays/vlen-arrays of predefined types, not " +
122 to_string(common));
123 return H5Buffer();
124 }
125 }

131 SmartBuffer ParallelMerger<Operator, Identity, Condition>::Functor<T>::operator()(
132 std::size_t n, const std::vector<const void *> &buffers) {
133 SmartBuffer outBuffer = SmartBuffer::copyValue(Identity<T>::identity(), n);
134 T *out = static_cast<T *>(outBuffer.get());
135 Operator<T> op;
136 for (const void *inBuffer : buffers) {
137 const T *buffer = static_cast<const T *>(inBuffer);
138
139 for (std::size_t idx = 0; idx < n; ++idx)
140 out[idx] = op(out[idx], buffer[idx]);
141 }
142 return outBuffer;
143 }
