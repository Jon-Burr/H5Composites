48 template <typename T>

50 struct BufferWriteTraits<T> {
51
52 static constexpr inline bool memcpy = true;
53
54 static void write(const UnderlyingType_t<T> &t, H5BufferView buffer) {
55 const H5::DataType &sourceDType = getH5DType<T>(t);
56 if (sourceDType == buffer.dtype())
57 std::memcpy(buffer.get(), &t, sizeof(UnderlyingType_t<T>));
58 else {
59 H5Buffer converted = convert(H5BufferConstView(&t, sourceDType), buffer.dtype());
60 std::memcpy(buffer.get(), converted.get(), buffer.footprint());
61 // The provider of the buffer pointer is also responsible for any variable length
62 // memory attached to it so we relinquish control over that
63 converted.transferVLenOwnership().release();
64 }
65 }
66 };
