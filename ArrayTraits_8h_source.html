50 template <typename U = T>
51 static std::enable_if_t<has_static_h5dtype_v<U>, std::array<U, N>> read(
52 const void *buffer, const H5::DataType &dtype) {
53 const H5::DataType &targetDType = getH5DType<std::array<U, N>>();
54 if (dtype == targetDType)
55 return *reinterpret_cast<const std::array<U, N> *>(buffer);

61 template <typename U = T>
62 static std::enable_if_t<!has_static_h5dtype_v<U>, std::array<UnderlyingType_t<U>, N>> read(
63 const void *buffer, const H5::DataType &dtype) {
64 // Check that the size is correct. Let H5 throw the error if it's not a CompType
65 H5::CompType compDType(dtype.getId());
66 if (compDType.getNmembers() != N)
67 throw std::invalid_argument(
68 "Incorrect number of elements in data type: " +
69 std::to_string(compDType.getNmembers()) + " vs " + std::to_string(N) +
70 " expected");
71 std::array<UnderlyingType_t<U>, N> arr;
72 readRangeFromCompoundDType(buffer, compDType, arr.begin());
73 return arr;
74 }
