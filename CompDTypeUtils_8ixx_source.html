12 template <WithH5DType T, std::ranges::input_range Range>
13 H5::CompType getCompoundDTypeFromRange(Range &&range) {
14 // First iterate through and get all of the data types
15 std::size_t totalSize = 0;
16 std::vector<H5::DataType> dtypes;
17 for (const UnderlyingType_t<T> &value : range) {
18 H5::DataType dtype = getH5DType<T>(value);
19 totalSize += dtype.getSize();
20 dtypes.push_back(dtype);
21 }
22
23 H5::CompType dtype(totalSize);
24 std::size_t offset = 0;
25 for (std::size_t idx = 0; idx < dtypes.size(); ++idx) {
26 dtype.insertMember("element" + std::to_string(idx), offset, dtypes.at(idx));
27 offset += dtypes.at(idx).getSize();
28 }
29 return dtype;
30 }
